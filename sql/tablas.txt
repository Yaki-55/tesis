-- TABLAS NES
CREATE TABLE nes_periodos(id_periodo int, periodo varchar(255));
CREATE TABLE nes_carreras(id_carrera int, carrera varchar(255));
CREATE TABLE nes_grupos(id_periodo int, id_carrera int, id_grupo int, grupo varchar(255), semestre int);
CREATE TABLE nes_materias(id_materia int, id_periodo int, id_carrera int, id_grupo int, materia varchar(255));
CREATE TABLE nes_calificaciones (matricula int, id_grupo int, id_materia int, p1 float, a1 float, p2 float, a2 float, p3 float, a3 float, o float, oa float, pf float, pa float, e1 float, e2 float, esp float);
CREATE TABLE nes_estudiantes(matricula int, grupo varchar(255), id_grupo int, nombre varchar(255));

-- TABLAS TUTORIAS

CREATE TABLE carreras (codigoCarrera text NOT NULL, nombre text NOT NULL, codigoInstituto text NOT NULL, siglas text NOT NULL);

CREATE TABLE institutos (codigo text NOT NULL, nombre text NOT NULL);

CREATE TABLE materias (idplan bigint NOT NULL, nombre text NOT NULL, semestre int NOT NULL, id bigint NOT NULL);

CREATE TABLE periodo (id int NOT NULL, actual SMALLINT NOT NULL, nombre text NOT NULL, fechainicio text NOT NULL, fechafin text NOT NULL, avance1 date NOT NULL, avance2 date NOT NULL, avance3 date NOT NULL, avance4 date NOT NULL, anyo int NOT NULL);

CREATE TABLE planes (codigoCarrera text NOT NULL, nombre text NOT NULL, id bigint NOT NULL);
